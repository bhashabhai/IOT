from machine import Pin
import time

# Use GP17 pin on Pi Pico
LED_PIN = 17
led = Pin(LED_PIN, Pin.OUT)

try:
    while True:
        led.value(1)   # LED ON
        time.sleep(1)  # Wait 1 second
        led.value(0)   # LED OFF
        time.sleep(1)  # Wait 1 second
except KeyboardInterrupt:
    print("Program stopped")
